bool SupZero(int Number) {

}

int CountNumberOfFrequency(int NumberOfDijit,int Number) {
    
    int q = 0, r = 0, CountNumberFrequency = 0;
    while (SupZero(Number)) {

        q = Number / 10;
        r = Number % 10;
        if (r == NumberOfDijit)
            CountNumberFrequency++;
        Number = q;

    }

    return CountNumberFrequency;
}
